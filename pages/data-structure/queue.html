public boolean enqueue(E e){
    if (size == elements.length) return false;
    elements[putIdx++] = e;
    if (putIdx==elements.length) putIdx = 0;
    size++;
    return true;
}

public E dequeue(){
    if (size == 0) return null;
    E e = (E) elements[takeIdx++];
    if (takeIdx==elements.length) takeIdx = 0;
    size--;
    return e;
}
